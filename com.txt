$UserList = Get-Content ".\CleanUsers.txt"
$Searcher = New-Object DirectoryServices.DirectorySearcher
$Searcher.PageSize = 1000

foreach ($User in $UserList) {
    $Searcher.Filter = "(&(objectCategory=person)(objectClass=user)(sAMAccountName=$User))"
    $Searcher.PropertiesToLoad.Clear()
    $Searcher.PropertiesToLoad.Add("samaccountname") > $null
    $Searcher.PropertiesToLoad.Add("description") > $null

    $result = $Searcher.FindOne()

    if ($result -ne $null) {
        $sam = $result.Properties.samaccountname[0]
        $desc = $result.Properties.description

        if ($desc -and $desc[0].Trim() -ne "") {
            Write-Host "$sam : $($desc[0])"
        }
    }
}
